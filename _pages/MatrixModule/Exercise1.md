          def average_columns(A):
              """
              Compute the average of each column of a matrix using only
              matrix multiplication and no loops

              Parameters
              ----------
              A: ndarray(M, N)
              A matrix

              Returns
              -------
              ndarray(M)
              A column which is the average of all other columns
              """
              M = A.shape[0]
              N = A.shape[1]
              v = get_special_vector(A)
              return A.dot(v)
